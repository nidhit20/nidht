import requests import os def download_image(url, save_path): # Send a GET request to the URL to get the image data response = requests.get(url) if response.status_code == 200: # Save the image to the specified path with open(save_path, 'wb') as file: file.write(response.content) print("Image saved successfully to:", save_path) else: print("Failed to download image. Status code:", response.status_code) def main(): # Desktop path desktop_path = os.path.join(os.path.join(os.environ['USERPROFILE']), 'Desktop') # URL of the image image_url = "https://source.unsplash.com/random/" # File path to save the image file_name = "random_image.jpg" save_path = os.path.join(desktop_path, file_name) # Download and save the image download_image(image_url, save_path) if __name__ == "__main__": main()
